usage

server.js


    var worldConfiguration = require('./path/to/world.json');
    var blocker = require('blocker');

    var port = 8000;
    var parser = blocker.generator.parser;
    var worldInstance = parser.parse(worldConfiguration);

    var server = blocker.createServer(port, worldInstance);
    server.start();

        var url = 'http://localhost:8000';
